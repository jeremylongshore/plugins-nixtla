- Generates visualization plots and performance metrics

### How it works

### Why it's beneficial

- **Reduces time-to-first-forecast** from hours to minutes

```python
"""
TimeGPT Quickstart Pipeline - CONCEPTUAL EXAMPLE
Generated by Claude Code Plugin for Nixtla (Intent Solutions io)
Dataset: sales.csv
Forecast horizon: 30 days

IMPORTANT: This is a conceptual example showing the structure of generated code.
Requires valid NIXTLA_API_KEY environment variable to run.
Not production-ready without proper error handling and testing.
"""

from nixtla import NixtlaClient
import pandas as pd
import numpy as np
from pathlib import Path
import matplotlib.pyplot as plt
from datetime import datetime
import logging

# Configure logging
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger(__name__)

def load_and_validate_data(filepath: str) -> pd.DataFrame:
    """Load data with comprehensive validation"""
    logger.info(f"Loading data from {filepath}")

    df = pd.read_csv(filepath, parse_dates=['ds'])

    # Validate required columns
    required_cols = ['ds', 'y']
    missing = set(required_cols) - set(df.columns)
    if missing:
        raise ValueError(f"Missing required columns: {missing}")

    # Check for missing values
    if df['y'].isna().any():
        logger.warning("Found missing values in target column, forward-filling")
        df['y'] = df['y'].fillna(method='ffill')

    # Ensure chronological order
    df = df.sort_values('ds')

    # Detect frequency
    freq = pd.infer_freq(df['ds'])
    logger.info(f"Detected frequency: {freq}")

    return df, freq

def generate_forecast(df: pd.DataFrame, horizon: int = 30) -> pd.DataFrame:
    """Generate TimeGPT forecast with confidence intervals"""
    logger.info("Initializing TimeGPT client")

    # SAFETY: Never hardcode API keys. Always use environment variables
    api_key = os.environ.get("NIXTLA_API_KEY")
    if not api_key:
        raise ValueError("NIXTLA_API_KEY environment variable not set")

    client = NixtlaClient(
        api_key=api_key,
        max_retries=3,
        timeout=30
    )

    logger.info(f"Generating {horizon}-period forecast")

    forecast_df = client.forecast(
        df=df,
        time_col='ds',
        target_col='y',
        h=horizon,
        level=[80, 95],  # Confidence levels
        add_history=True,  # Include historical data in output
    )

    # Add metadata
    forecast_df['model'] = 'TimeGPT'
    forecast_df['generated_at'] = datetime.now()

    return forecast_df

def calculate_metrics(historical: pd.DataFrame, forecast: pd.DataFrame) -> dict:
    """Calculate performance metrics if test data available"""
    # Implementation for backtesting metrics
    metrics = {
        'forecast_horizon': len(forecast),
        'historical_points': len(historical),
        'forecast_start': forecast['ds'].min(),
        'forecast_end': forecast['ds'].max()
    }
    return metrics

def create_visualizations(historical: pd.DataFrame, forecast: pd.DataFrame):
    """Generate forecast plots"""
    fig, ax = plt.subplots(figsize=(12, 6))

    # Plot historical data
    ax.plot(historical['ds'], historical['y'],
            label='Historical', color='black', alpha=0.7)

    # Plot forecast with confidence intervals
    ax.plot(forecast['ds'], forecast['TimeGPT'],
            label='Forecast', color='blue', linewidth=2)
    ax.fill_between(
        forecast['ds'],
        forecast['TimeGPT-lo-95'],
        forecast['TimeGPT-hi-95'],
        alpha=0.2, color='blue', label='95% CI'
    )

    ax.set_xlabel('Date')
    ax.set_ylabel('Value')
    ax.set_title('TimeGPT Forecast Results')
    ax.legend()
    plt.tight_layout()

    # Save plot
    output_dir = Path('outputs')
    output_dir.mkdir(exist_ok=True)
    plt.savefig(output_dir / 'forecast_plot.png', dpi=300)
    logger.info("Visualization saved to outputs/forecast_plot.png")

def main():
    """Main pipeline execution"""
    try:
        # Load and validate data
        df, frequency = load_and_validate_data('data/sales.csv')
        logger.info(f"Loaded {len(df)} records")

        # Generate forecast
        forecast = generate_forecast(df, horizon=30)

        # Calculate metrics
        metrics = calculate_metrics(df, forecast)
        logger.info(f"Metrics: {metrics}")

        # Create visualizations
        create_visualizations(df, forecast)

        # Save results
        output_path = Path('outputs/forecast_results.csv')
        forecast.to_csv(output_path, index=False)
        logger.info(f"Results saved to {output_path}")

    except Exception as e:
        logger.error(f"Pipeline failed: {e}")
        raise

if __name__ == "__main__":
    import os
    from dotenv import load_dotenv

    load_dotenv()
    main()
```

## 2) Nixtla Bench Harness Generator

![Status](https://img.shields.io/badge/Status-Initial%20Concept%20·%20v0.1-blue)

Create comprehensive benchmark harnesses that compare multiple Nixtla models (StatsForecast, MLForecast, NeuralForecast, TimeGPT) on your specific dataset. Make data-driven model selection decisions with reproducible comparisons.

### What it does

Scaffolds a complete benchmarking framework that:
- Runs multiple Nixtla models on the same dataset
- Computes standardized metrics (MAPE, RMSE, MAE, MASE)
- Handles different model requirements gracefully
- Generates comparative reports in markdown and HTML
- Tracks computational resources (time, memory)
- Produces publication-ready comparison tables and charts

### How it works

The plugin guides you through benchmark configuration:
1. Select which models to compare (checkboxes for each Nixtla library)
2. Specify evaluation methodology (cross-validation folds, test set size)
3. Choose metrics to compute
4. Configure model-specific parameters if needed
5. Generates `bench_nixtla_models.py` with all selected comparisons
6. Creates a `configs/` directory with model configurations

### Why it's beneficial

- **Objective model selection** based on your specific data characteristics
- **Reproducible experiments** with versioned configurations
- **Fair comparisons** using consistent preprocessing and evaluation
- **Resource awareness** to understand computational tradeoffs
- **Automated reporting** for stakeholder communication

```python
"""
Nixtla Model Benchmark Harness - CONCEPTUAL EXAMPLE
Compares TimeGPT, StatsForecast, MLForecast, and NeuralForecast
Generated by Claude Code Plugin for Nixtla (Intent Solutions io)

IMPORTANT: Conceptual demonstration only. Requires:
- Valid NIXTLA_API_KEY environment variable
- All Nixtla libraries installed (pip install nixtla statsforecast mlforecast neuralforecast)
- Sufficient compute resources for neural models
- Proper data in expected format

TODO: Add production error handling, resource limits, and timeout management
"""

import pandas as pd
import numpy as np
from typing import Dict, List, Tuple
from dataclasses import dataclass
import time
import psutil
import json
from pathlib import Path

# Nixtla imports
from nixtla import NixtlaClient
from statsforecast import StatsForecast
from statsforecast.models import AutoARIMA, ETS, Theta
from mlforecast import MLForecast
from mlforecast.models import LinearRegression, XGBRegressor, LGBMRegressor
from neuralforecast import NeuralForecast
from neuralforecast.models import NBEATS, NHITS, TFT

# Metrics
from sklearn.metrics import mean_absolute_error, mean_squared_error
from statsforecast.utils import mase

@dataclass
class BenchmarkResult:
    """Store results for each model"""
    model_name: str
    model_type: str  # statistical, ml, neural, foundation
    predictions: pd.DataFrame
    metrics: Dict[str, float]
    training_time: float
    inference_time: float
    memory_usage: float
    parameters: Dict

class NixtlaBenchmark:
    """Comprehensive benchmark harness for Nixtla models"""

    def __init__(self, data_path: str, horizon: int = 24):
        self.data = pd.read_csv(data_path, parse_dates=['ds'])
        self.horizon = horizon
        self.results: List[BenchmarkResult] = []
        self.train, self.test = self._split_data()

    def _split_data(self) -> Tuple[pd.DataFrame, pd.DataFrame]:
        """Split data into train/test sets"""
        cutoff = len(self.data) - self.horizon
        return self.data[:cutoff], self.data[cutoff:]

    def _compute_metrics(self, y_true: np.ndarray, y_pred: np.ndarray) -> Dict[str, float]:
        """Calculate all evaluation metrics"""
        return {
            'mae': mean_absolute_error(y_true, y_pred),
            'rmse': np.sqrt(mean_squared_error(y_true, y_pred)),
            'mape': np.mean(np.abs((y_true - y_pred) / y_true)) * 100,
            'mase': mase(y_true, y_pred, y_train=self.train['y'].values),
        }

    def benchmark_timegpt(self) -> BenchmarkResult:
        """Benchmark TimeGPT"""
        print("Benchmarking TimeGPT...")
        start_time = time.time()
        start_memory = psutil.Process().memory_info().rss / 1024 / 1024

        # SAFETY CHECK: Ensure API key is available
        api_key = os.environ.get("NIXTLA_API_KEY")
        if not api_key:
            print("WARNING: NIXTLA_API_KEY not set, skipping TimeGPT benchmark")
            return None

        client = NixtlaClient(api_key=api_key)

        # Training is implicit in TimeGPT
        train_time = time.time() - start_time

        # Inference
        inf_start = time.time()
        forecast = client.forecast(
            df=self.train,
            time_col='ds',
            target_col='y',
            h=self.horizon,
        )
        inference_time = time.time() - inf_start

        # Memory usage
        end_memory = psutil.Process().memory_info().rss / 1024 / 1024
        memory_used = end_memory - start_memory

        # Compute metrics
        metrics = self._compute_metrics(
            self.test['y'].values,
            forecast['TimeGPT'].values
        )

        return BenchmarkResult(
            model_name="TimeGPT",
            model_type="foundation",
            predictions=forecast,
            metrics=metrics,
            training_time=train_time,
            inference_time=inference_time,
            memory_usage=memory_used,
            parameters={'horizon': self.horizon}
        )

    def benchmark_statsforecast(self) -> BenchmarkResult:
        """Benchmark StatsForecast models"""
        print("Benchmarking StatsForecast...")
        start_time = time.time()
        start_memory = psutil.Process().memory_info().rss / 1024 / 1024

        # Configure models
        models = [
            AutoARIMA(season_length=7),
            ETS(season_length=7),
            Theta()
        ]

        sf = StatsForecast(
            models=models,
            freq='D',
            n_jobs=-1
        )

        # Training
        sf.fit(self.train[['ds', 'y', 'unique_id']])
        train_time = time.time() - start_time

        # Inference
        inf_start = time.time()
        forecast = sf.predict(h=self.horizon)
        inference_time = time.time() - inf_start

        # Use ensemble average
        y_pred = forecast[['AutoARIMA', 'ETS', 'Theta']].mean(axis=1).values

        # Memory and metrics
        end_memory = psutil.Process().memory_info().rss / 1024 / 1024
        metrics = self._compute_metrics(self.test['y'].values, y_pred)

        return BenchmarkResult(
            model_name="StatsForecast Ensemble",
            model_type="statistical",
            predictions=forecast,
            metrics=metrics,
            training_time=train_time,
            inference_time=inference_time,
            memory_usage=end_memory - start_memory,
            parameters={'models': ['AutoARIMA', 'ETS', 'Theta']}
        )

    def benchmark_mlforecast(self) -> BenchmarkResult:
        """Benchmark MLForecast"""
        print("Benchmarking MLForecast...")
        start_time = time.time()
        start_memory = psutil.Process().memory_info().rss / 1024 / 1024

        # Configure model with features
        mlf = MLForecast(
            models=[
                LinearRegression(),
                XGBRegressor(random_state=42),
                LGBMRegressor(random_state=42)
            ],
            freq='D',
            lags=[7, 14, 21],
            lag_transforms={
                1: [(lambda x: x.rolling(window=7).mean(), 'rolling_mean_7')],
                7: [(lambda x: x.rolling(window=7).std(), 'rolling_std_7')]
            },
            date_features=['dayofweek', 'month']
        )

        # Training
        mlf.fit(self.train[['ds', 'y', 'unique_id']])
        train_time = time.time() - start_time

        # Inference
        inf_start = time.time()
        forecast = mlf.predict(h=self.horizon)
        inference_time = time.time() - inf_start

        # Ensemble predictions
        model_cols = ['LinearRegression', 'XGBRegressor', 'LGBMRegressor']
        y_pred = forecast[model_cols].mean(axis=1).values

        # Metrics
        end_memory = psutil.Process().memory_info().rss / 1024 / 1024
        metrics = self._compute_metrics(self.test['y'].values, y_pred)

        return BenchmarkResult(
            model_name="MLForecast Ensemble",
            model_type="ml",
            predictions=forecast,
            metrics=metrics,
            training_time=train_time,
            inference_time=inference_time,
            memory_usage=end_memory - start_memory,
            parameters={
                'models': model_cols,
                'lags': [7, 14, 21],
                'date_features': ['dayofweek', 'month']
            }
        )

    def benchmark_neuralforecast(self) -> BenchmarkResult:
        """Benchmark NeuralForecast"""
        print("Benchmarking NeuralForecast...")
        start_time = time.time()
        start_memory = psutil.Process().memory_info().rss / 1024 / 1024

        # Configure models (lighter versions for quick benchmark)
        models = [
            NBEATS(
                input_size=2 * self.horizon,
                h=self.horizon,
                max_steps=100
            ),
            NHITS(
                input_size=2 * self.horizon,
                h=self.horizon,
                max_steps=100
            )
        ]

        nf = NeuralForecast(models=models, freq='D')

        # Training
        nf.fit(df=self.train[['ds', 'y', 'unique_id']])
        train_time = time.time() - start_time

        # Inference
        inf_start = time.time()
        forecast = nf.predict()
        inference_time = time.time() - inf_start

        # Average predictions
        y_pred = forecast[['NBEATS', 'NHITS']].mean(axis=1).values

        # Metrics
        end_memory = psutil.Process().memory_info().rss / 1024 / 1024
        metrics = self._compute_metrics(self.test['y'].values, y_pred)

        return BenchmarkResult(
            model_name="NeuralForecast Ensemble",
            model_type="neural",
            predictions=forecast,
            metrics=metrics,
            training_time=train_time,
            inference_time=inference_time,
            memory_usage=end_memory - start_memory,
            parameters={'models': ['NBEATS', 'NHITS'], 'max_steps': 100}
        )

    def run_all_benchmarks(self):
        """Execute all configured benchmarks"""
        # Add unique_id column if missing (required by some models)
        if 'unique_id' not in self.train.columns:
            self.train['unique_id'] = 0
            self.test['unique_id'] = 0

        benchmarks = [
            self.benchmark_timegpt,
            self.benchmark_statsforecast,
            self.benchmark_mlforecast,
            self.benchmark_neuralforecast
        ]

        for benchmark_fn in benchmarks:
            try:
                result = benchmark_fn()
                self.results.append(result)
            except Exception as e:
                print(f"Failed to run {benchmark_fn.__name__}: {e}")

    def generate_report(self):
        """Generate comparison report"""
        # Create results DataFrame
        comparison = pd.DataFrame([
            {
                'Model': r.model_name,
                'Type': r.model_type,
                'MAE': f"{r.metrics['mae']:.3f}",
                'RMSE': f"{r.metrics['rmse']:.3f}",
                'MAPE': f"{r.metrics['mape']:.2f}%",
                'MASE': f"{r.metrics['mase']:.3f}",
                'Train Time (s)': f"{r.training_time:.2f}",
                'Inference Time (s)': f"{r.inference_time:.2f}",
                'Memory (MB)': f"{r.memory_usage:.1f}"
            }
            for r in self.results
        ])

        # Save as markdown
        output_dir = Path('benchmark_results')
        output_dir.mkdir(exist_ok=True)

        with open(output_dir / 'report.md', 'w') as f:
            f.write("# Nixtla Model Benchmark Results\n\n")
            f.write(f"Dataset: {len(self.data)} observations\n")
            f.write(f"Forecast Horizon: {self.horizon} periods\n\n")
            f.write("## Performance Comparison\n\n")
            f.write(comparison.to_markdown(index=False))
            f.write("\n\n## Best Model by Metric\n\n")

            # Find best for each metric
            for metric in ['MAE', 'RMSE', 'MAPE', 'MASE']:
                col_values = comparison[metric].str.rstrip('%').astype(float)
                best_idx = col_values.idxmin()
                best_model = comparison.loc[best_idx, 'Model']
                f.write(f"- **{metric}**: {best_model} ({comparison.loc[best_idx, metric]})\n")

        print(f"\nBenchmark complete! Report saved to {output_dir / 'report.md'}")
        return comparison

if __name__ == "__main__":
    import os
    from dotenv import load_dotenv

    load_dotenv()

    # Run benchmark
    bench = NixtlaBenchmark('data/sales.csv', horizon=30)
    bench.run_all_benchmarks()
    results = bench.generate_report()
    print("\n", results)
```

```python
class TimeSeriesData(BaseModel):
    """Input time series data"""
    ds: List[str] = Field(..., description="Timestamps in ISO format")
    y: List[float] = Field(..., description="Target values")

    @validator('ds')
    def validate_timestamps(cls, v):
        """Ensure timestamps are parseable"""
        try:
            [pd.Timestamp(ts) for ts in v]
        except:

        return v

    @validator('y')
    def validate_values(cls, v, values):

            raise ValueError("ds and y must have same length")
        return v

@app.post("/forecast", response_model=ForecastResponse)
async def generate_forecast(
    request: ForecastRequest,
    background_tasks: BackgroundTasks,
    api_key: str = Depends(verify_api_key)
):
    """Generate time series forecast"""
    start_time = asyncio.get_event_loop().time()

    # Create cache key from request
    cache_key = hashlib.md5(
        f"{request.data.json()}:{request.horizon}:{request.model}".encode()
    ).hexdigest()

    # Check cache
    cached_result = redis_client.get(f"forecast:{cache_key}")
    if cached_result:
        result = json.loads(cached_result)
        result["cached"] = True
        result["execution_time_ms"] = (asyncio.get_event_loop().time() - start_time) * 1000
        return ForecastResponse(**result)

    try:
        # Convert to DataFrame
        df = pd.DataFrame({
            'ds': pd.to_datetime(request.data.ds),
            'y': request.data.y
        })

        # Detect frequency if not provided
        if not request.frequency:
            request.frequency = pd.infer_freq(df['ds'])
            logger.info(f"Detected frequency: {request.frequency}")

        # Generate forecast based on model selection
        if request.model == "timegpt":
            forecast_df = await _forecast_timegpt(df, request)
        elif request.model == "statsforecast":
            forecast_df = await _forecast_statsforecast(df, request)
        else:
            # Auto-select based on data characteristics
            forecast_df = await _auto_select_and_forecast(df, request)

        # Format response
        forecast_data = []
        for _, row in forecast_df.iterrows():
            forecast_point = {
                "ds": row['ds'].isoformat(),
                "forecast": row['forecast'],
            }

            # Add confidence intervals
            for level in request.level:
                if f'lo-{level}' in row:
                    forecast_point[f'lower_{level}'] = row[f'lo-{level}']
                    forecast_point[f'upper_{level}'] = row[f'hi-{level}']

            forecast_data.append(forecast_point)

        # Create response
        response_data = {
            "status": "success",
            "model_used": request.model,
            "forecast": forecast_data,
            "metrics": {
                "horizon": request.horizon,
                "frequency": request.frequency,
                "input_points": len(df)
            },
            "cached": False,
            "generated_at": datetime.now(),
            "execution_time_ms": (asyncio.get_event_loop().time() - start_time) * 1000
        }

        # Cache result (1 hour TTL)
        background_tasks.add_task(
            cache_forecast,
            cache_key,
            response_data
        )

        return ForecastResponse(**response_data)

    except Exception as e:
        logger.error(f"Forecast failed: {e}")
        raise HTTPException(status_code=500, detail=str(e))

async def _forecast_timegpt(df: pd.DataFrame, request: ForecastRequest) -> pd.DataFrame:
    """Generate forecast using TimeGPT"""
    client = app.state.nixtla_client

    # Run in thread pool to avoid blocking
    loop = asyncio.get_event_loop()
    forecast = await loop.run_in_executor(
        None,
        lambda: client.forecast(
            df=df,
            time_col='ds',
            target_col='y',
            h=request.horizon,
            level=request.level
        )
    )

    # Rename columns for consistency
    forecast = forecast.rename(columns={'TimeGPT': 'forecast'})
    for level in request.level:
        forecast = forecast.rename(columns={
            f'TimeGPT-lo-{level}': f'lo-{level}',
            f'TimeGPT-hi-{level}': f'hi-{level}'
        })

    return forecast

async def _forecast_statsforecast(df: pd.DataFrame, request: ForecastRequest) -> pd.DataFrame:
    """Generate forecast using StatsForecast"""
    # Add required unique_id column
    df['unique_id'] = 0

    # Initialize model
    model = AutoARIMA(season_length=7)  # Adjust based on frequency
    sf = StatsForecast(models=[model], freq=request.frequency or 'D')

    # Run in thread pool
    loop = asyncio.get_event_loop()
    sf = await loop.run_in_executor(None, sf.fit, df)
    forecast = await loop.run_in_executor(
        None,
        lambda: sf.predict(h=request.horizon, level=request.level)
    )

    # Format output
    forecast = forecast.reset_index()
    forecast = forecast.rename(columns={'AutoARIMA': 'forecast'})

    return forecast

async def _auto_select_and_forecast(df: pd.DataFrame, request: ForecastRequest) -> pd.DataFrame:
    """Auto-select best model based on data characteristics"""
    # Simple heuristic: use TimeGPT for longer series, StatsForecast for shorter
    if len(df) > 100:
        return await _forecast_timegpt(df, request)
    else:
        return await _forecast_statsforecast(df, request)

def cache_forecast(cache_key: str, data: dict):
    """Background task to cache forecast results"""
    redis_client.setex(
        f"forecast:{cache_key}",
        3600,  # 1 hour TTL
        json.dumps(data, default=str)
    )
    logger.info(f"Cached forecast with key: {cache_key}")
```
